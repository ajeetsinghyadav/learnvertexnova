CMake Build System

Reference and notes for CMake — the cross-platform build system used by VertexNova and most C/C++ projects.

Overview​

CMake is a meta-build system. It reads CMakeLists.txt and generates native build files (Makefiles, Ninja, Visual Studio, Xcode). You then run the native tool to compile.

StepCommandDoes
Configurecmake -B buildGenerate build files
Buildcmake --build buildCompile the project
Installcmake --install buildCopy artifacts to prefix
Two-stage process

Always keep configure and build separate. Never run cmake . in source dir — always cmake -B build (out-of-source).

Core Concepts​

  • Target — Executable or library. Carries its own compile flags, includes, and link deps.
  • CMakeLists.txt — Build definition. Every directory can have one. Root is entry point.
  • Generator — What CMake outputs: Ninja, Makefiles, Visual Studio, Xcode. Set with -G.
  • Build tree — Out-of-source dir (e.g. build/). Never build in-source.
  • Cache — CMakeCache.txt stores config. Override with -D flags.

Project Structure​

my-project/
├── CMakeLists.txt          # project() + subdirs
├── src/
│   ├── CMakeLists.txt      # add_executable / add_library
│   └── main.cpp
├── include/
│   └── renderer.h
├── libs/core/
│   └── CMakeLists.txt      # library target
└── build/                  # OUT-OF-SOURCE — git-ignored

Minimal root CMakeLists.txt​

cmake_minimum_required(VERSION 3.20)
project(MyProject VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_subdirectory(src)
add_subdirectory(libs/core)

Targets & Linking​

Creating targets​

add_executable(myapp src/main.cpp src/app.cpp)

add_library(mylib STATIC src/renderer.cpp)    # static .a / .lib
add_library(mylib SHARED src/plugin.cpp)      # shared .so / .dll
add_library(mylib INTERFACE)                  # header-only, no sources

target_* — the modern way​

target_include_directories(mylib
  PUBLIC  ${CMAKE_CURRENT_SOURCE_DIR}/include   # consumers get it too
  PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src       # only this target
)

target_link_libraries(myapp
  PRIVATE   mylib      # link, don't expose to consumers
  PUBLIC    vne_math   # link AND expose (transitive)
)

target_compile_options(mylib PRIVATE -Wall -Wextra)

target_compile_definitions(mylib PRIVATE VNE_VERSION=1)

PRIVATE / PUBLIC / INTERFACE​

KeywordThis targetConsumers (linking targets)
PRIVATE✓ applied✗ not propagated
PUBLIC✓ applied✓ propagated
INTERFACE✗ not applied✓ propagated

Variables & Options​

Variables​

set(MY_SOURCES src/a.cpp src/b.cpp)
add_library(mylib ${MY_SOURCES})
VariableMeaning
CMAKE_SOURCE_DIRRoot source dir
CMAKE_CURRENT_SOURCE_DIRCurrent CMakeLists.txt dir
CMAKE_BINARY_DIRRoot build dir
CMAKE_BUILD_TYPEDebug | Release | RelWithDebInfo
CMAKE_INSTALL_PREFIXInstall destination
PROJECT_SOURCE_DIRNearest project() source dir

Options​

option(VNE_BUILD_TESTS "Build tests" OFF)
option(VNE_ENABLE_VULKAN "Enable Vulkan" ON)

if(VNE_BUILD_TESTS)
  add_subdirectory(tests)
endif()

Override: cmake -B build -DVNE_BUILD_TESTS=ON

Generator expressions​

Evaluated at build time. Syntax: $<condition:value>

target_compile_options(myapp PRIVATE
  $<$<CONFIG:Debug>:-g -O0>
  $<$<CONFIG:Release>:-O3 -DNDEBUG>
)

target_compile_definitions(myapp PRIVATE
  $<$<PLATFORM_ID:Windows>:NOMINMAX>
)

BUILD_INTERFACE / INSTALL_INTERFACE​

Use different include paths for build (source tree) vs install (deployed). Essential when packaging libraries.

target_include_directories(mylib PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>
)
  • BUILD_INTERFACE — Used when consumers build against your source (e.g. add_subdirectory). Path is absolute.
  • INSTALL_INTERFACE — Used when consumers use installed headers. Path is relative to CMAKE_INSTALL_PREFIX.

Library alias​

Create an alias target so consumers link to a stable name (e.g. mylib::mylib) instead of the raw target.

add_library(mylib STATIC src/mylib.cpp)
add_library(mylib::mylib ALIAS mylib)

# Consumers link to mylib::mylib — same as mylib but namespaced
target_link_libraries(myapp PRIVATE mylib::mylib)

Useful for consistency with imported targets (Vulkan::Vulkan) and install/export.

find_package & FetchContent​

find_package (system-installed)​

find_package(Vulkan REQUIRED)
find_package(OpenGL REQUIRED)
find_package(glfw3 3.3 REQUIRED)

target_link_libraries(myapp PRIVATE
  Vulkan::Vulkan    # imported target — brings includes + libs
  OpenGL::GL
  glfw
)

# Optional dependency
find_package(Catch2 QUIET)
if(Catch2_FOUND)
  add_subdirectory(tests)
endif()

FetchContent (download at configure)​

include(FetchContent)

FetchContent_Declare(
  glm
  GIT_REPOSITORY https://github.com/g-truc/glm.git
  GIT_TAG        0.9.9.8
)

FetchContent_MakeAvailable(glm)
target_link_libraries(myapp PRIVATE glm::glm)
When to use which

find_package — System-installed (Vulkan SDK, OpenGL). FetchContent — Pin a version, header-only or small libs.

Google Test​

Option A — FetchContent (download at configure):

include(FetchContent)

FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG        v1.14.0
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

add_executable(my_test tests/my_test.cpp)
target_link_libraries(my_test PRIVATE GTest::gtest_main)

Option B — Submodule in deps (bundled):

# Googletest in deps/external/googletest (git submodule)
message(STATUS "Using Google Test from deps/external/googletest (v1.17.0)")
add_subdirectory(
  ${CMAKE_SOURCE_DIR}/deps/external/googletest
  ${CMAKE_BINARY_DIR}/deps/external/googletest
)
include(GoogleTest)

set(GTEST_LIBRARIES gtest_main gmock)
set(GTEST_INCLUDE_DIRS
  ${CMAKE_SOURCE_DIR}/deps/external/googletest/googletest/include
  ${CMAKE_SOURCE_DIR}/deps/external/googletest/googlemock/include
)

# Use in tests
add_executable(my_test tests/my_test.cpp)
target_link_libraries(my_test PRIVATE ${GTEST_LIBRARIES})
target_include_directories(my_test PRIVATE ${GTEST_INCLUDE_DIRS})

Install rules (third-party / packaging)​

Define what gets installed when running cmake --install build.

# Install library
install(TARGETS mylib
  LIBRARY DESTINATION lib
  ARCHIVE DESTINATION lib
  RUNTIME DESTINATION bin
  INCLUDES DESTINATION include
)

# Install headers
install(DIRECTORY include/
  DESTINATION include
  FILES_MATCHING PATTERN "*.h" PATTERN "*.hpp"
)

# Install config for find_package (advanced)
install(EXPORT mylib-targets
  FILE mylib-targets.cmake
  NAMESPACE mylib::
  DESTINATION lib/cmake/mylib
)

Commands​

Configure​

cmake -B build
cmake -B build -G "Ninja"
cmake -B build -DCMAKE_BUILD_TYPE=Release
cmake -B build -DCMAKE_INSTALL_PREFIX=./dist
cmake -B build -DCMAKE_TOOLCHAIN_FILE=cmake/arm.cmake
cmake --preset=release

Build & install​

cmake --build build
cmake --build build --parallel 8
cmake --build build --target mylib
cmake --build build --config Release    # MSVC/Xcode
cmake --install build
cmake --install build --prefix ./dist
cmake --build build --target clean

Testing (CTest)​

enable_testing()

add_executable(renderer_tests tests/renderer_test.cpp)
target_link_libraries(renderer_tests PRIVATE mylib Catch2::Catch2WithMain)

add_test(NAME RendererTests COMMAND renderer_tests)
ctest --test-dir build --output-on-failure
ctest --test-dir build -V --parallel 4
ctest --test-dir build -R Renderer

Modern CMake Tips​

Avoid legacy patterns

include_directories(), link_libraries(), add_compile_options() (without target_), and CMAKE_CXX_FLAGS manipulation. Use target_* always.

  • target_ always* — Every include, flag, define via target_* — never global.
  • Out-of-source build — Always cmake -B build. Add build/ to .gitignore.
  • Prefer Ninja — Faster than Make, works everywhere.
  • Imported targets — Link to Vulkan::Vulkan, not VULKAN_LIBRARIES.
  • cmake_minimum_required first — Very first line. Use 3.20+ for new projects.

CMakePresets.json​

{
  "version": 6,
  "configurePresets": [
    {
      "name": "release",
      "generator": "Ninja",
      "binaryDir": "${sourceDir}/build/release",
      "cacheVariables": {
        "CMAKE_BUILD_TYPE": "Release",
        "VNE_ENABLE_VULKAN": "ON"
      }
    }
  ]
}
cmake --preset=release
cmake --build --preset=release
